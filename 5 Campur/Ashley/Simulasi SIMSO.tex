\begin{enumerate}

    % Answer: 2 \cdot 3^2 \cdot 5^2 \cdot 7
    \item Find the prime factorization of 3150.
    \begin{enumerate}
        \item $2 \cdot 3^2 \cdot 5 \cdot 7^2$
        \item $2^2 \cdot 3 \cdot 5^2 \cdot 7$
        \item $2 \cdot 3^2 \cdot 5^2 \cdot 7$
        \item $2 \cdot 3^3 \cdot 5 \cdot 7$
    \end{enumerate}

    % Answer: 24
    \item How many positive divisors does the number 360 have?
    \begin{enumerate}
        \item 18
        \item 20
        \item 24
        \item 30
    \end{enumerate}

    % Answer: k \le -8 \text{ or } k \ge 8
    \item For what values of $k$ does the equation $x^2 + kx + 16 = 0$ have real roots?
    \begin{enumerate}
        \item $-8 \le k \le 8$
        \item $k \le -8 \text{ or } k \ge 8$
        \item $k \le -4 \text{ or } k \ge 4$
        \item $k \ge 8 \text{ only}$
    \end{enumerate}

    % Answer: 5
    \item Find the sum of $x$ and $y$ if $2x + 3y = 12$ and $4x - y = 10$.
    \begin{enumerate}
        \item 3
        \item 4
        \item 5
        \item 6
    \end{enumerate}

    % Answer: 5x^2\sqrt{3x}
    \item Simplify the expression $\sqrt{75x^5}$.
    \begin{enumerate}
        \item $3x^2\sqrt{5x}$
        \item $5x^2\sqrt{3x}$
        \item $5x\sqrt{3x^2}$
        \item $15x^2\sqrt{x}$
    \end{enumerate}

    % Answer: 2.4 hours
    \item John can paint a room in 4 hours. Mary can paint the same room in 6 hours. How long will it take them to paint the room together?
    \begin{enumerate}
        \item 2 hours
        \item 2.4 hours
        \item 3 hours
        \item 5 hours
    \end{enumerate}

    % Answer: 11
    \item In a class of 40 students, 25 play soccer, 18 play basketball, and 8 play neither. How many students play both sports?
    \begin{enumerate}
        \item 9
        \item 10
        \item 11
        \item 12
    \end{enumerate}

    % Answer: 61
    \item Find the 15th term of the arithmetic sequence: $5, 9, 13, 17, \dots$
    \begin{enumerate}
        \item 57
        \item 61
        \item 65
        \item 69
    \end{enumerate}

    % Answer: 728
    \item What is the sum of the first 6 terms of the geometric sequence where the first term is 2 and the common ratio is 3?
    \begin{enumerate}
        \item 242
        \item 486
        \item 728
        \item 1456
    \end{enumerate}

    % Answer: 120 cm^2
    \item What is the area of a rhombus if its diagonals are 10 cm and 24 cm?
    \begin{enumerate}
        \item $60\text{ cm}^2$
        \item $120\text{ cm}^2$
        \item $240\text{ cm}^2$
        \item $480\text{ cm}^2$
    \end{enumerate}

    % Answer: 150^\circ
    \item The measures of the interior angles of a pentagon are in the ratio $2:3:4:4:5$. What is the measure of the largest angle?
    \begin{enumerate}
        \item $120^\circ$
        \item $135^\circ$
        \item $140^\circ$
        \item $150^\circ$
    \end{enumerate}

    % Answer: 60
    \item How many different 6-letter arrangements can be made using the letters of the word ``BANANA''?
    \begin{enumerate}
        \item 60
        \item 120
        \item 360
        \item 720
    \end{enumerate}

    % Answer: 210
    \item A committee of 4 people is to be chosen from a group of 10. How many different committees can be formed?
    \begin{enumerate}
        \item 120
        \item 210
        \item 5040
        \item 2520
    \end{enumerate}

    % Answer: 2912
    \item The sum of three consecutive even integers is 162. What is the product of the smallest and largest of these integers?
    \begin{enumerate}
        \item 2808
        \item 2912
        \item 3136
        \item 3024
    \end{enumerate}

    % Answer: \frac{5}{12}
    \item Two fair six-sided dice are rolled. What is the probability that the sum of the numbers rolled is a prime number?
    \begin{enumerate}
        \item $\frac{1}{3}$
        \item $\frac{5}{12}$
        \item $\frac{1}{2}$
        \item $\frac{7}{18}$
    \end{enumerate}

    % Answer: 2
    \item If the polynomial $P(x) = 2x^3 - kx^2 + 3x - 5$ is divided by $(x - 2)$, the remainder is 9. Find the value of $k$.
    \begin{enumerate}
        \item -2
        \item 0
        \item 2
        \item 4
    \end{enumerate}

    % Answer: 10
    \item Find the distance between the points $A(-3, 4)$ and $B(5, -2)$ on the coordinate plane.
    \begin{enumerate}
        \item 8
        \item $2\sqrt{17}$
        \item 10
        \item 14
    \end{enumerate}

    % Answer: 20
    \item What is the smallest positive integer $n$ such that $n!$ ends in exactly 4 trailing zeros?
    \begin{enumerate}
        \item 15
        \item 16
        \item 20
        \item 24
    \end{enumerate}

    % Answer: 26
    \item Let $f(x) = 3x - 4$ and $g(x) = x^2 + 1$. What is the value of $f(g(3))$?
    \begin{enumerate}
        \item 10
        \item 26
        \item 32
        \item 74
    \end{enumerate}

    % Answer: 4
    \item Two trains start from stations 400 km apart and travel towards each other. One travels at 45 km/h and the other at 55 km/h. How many hours will it take for them to meet?
    \begin{enumerate}
        \item 3.5
        \item 4
        \item 4.5
        \item 5
    \end{enumerate}

\end{enumerate}
