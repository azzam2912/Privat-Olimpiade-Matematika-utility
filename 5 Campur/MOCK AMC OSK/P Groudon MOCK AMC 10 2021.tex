\begin{enumerate}

\item Compute $2021-2^{0+1+2}+10\times21$

\quad (A) 2217 \quad (B) 2221 \quad (C) 2223 \quad (D) 2227 \quad (E) 2233

\item Suppose $n\%$ of $m$ is 4 and $m\%$ of $mn$ is 6. What is $n$?

\quad (A) $\frac{800}{3}$ \quad (B) 300 \quad (C) $\frac{625}{2}$ \quad (D) $\frac{1000}{3}$ \quad (E) 360

\item How many ways are there to arrange the numbers 1, 2, 3, 4 in a line so that the product of some two adjacent numbers in the line is odd?

\quad (A) 8 \quad (B) 12 \quad (C) 16 \quad (D) 18 \quad (E) 20

\item How many positive integers $a\le50$ satisfy $\gcd(a,50)=5?$

\quad (A) 2 \quad (B) 4 \quad (C) 5 \quad (D) 6 \quad (E) 9

\item Let $ABCDEF$ be a regular hexagon with side length 1. If $M$, $N$, and $P$ are the midpoints of $AB$, $BC$, and $CD$, respectively, what is the area of $\triangle MNP?$

\quad (A) $\frac{3\sqrt{3}}{16}$ \quad (B) $\frac{\sqrt{3}}{4}$ \quad (C) $\frac{3\sqrt{3}}{8}$ \quad (D) $\frac{\sqrt{3}}{2}$ \quad (E) $\frac{9\sqrt{3}}{16}$

\item For the smallest multiple of 11 that can be written as $ababa$, where $a$ and $b$ are digits and $a\ne0$ what is $a+b?$

\quad (A) 5 \quad (B) 6 \quad (C) 8 \quad (D) 10 \quad (E) 11

\item Four candles are each lit and arranged in a circle. Suddenly, a gust of wind blows and each candle has a 50\% chance of going out. What is the probability that after the gust, each candle is either still lit or is next to a lit candle?

\quad (A) $\frac{1}{2}$ \quad (B) $\frac{9}{16}$ \quad (C) $\frac{5}{8}$ \quad (D) $\frac{11}{16}$ \quad (E) $\frac{3}{4}$

\item A positive integer $N$ can be represented as $74_{a}$ in base-$a$, $61_{b}$ in base-$b$, and $52_{c}$ in base-$c$. The positive integers $a$, $b$, and $c$ form an arithmetic sequence in that order. What is the sum of the digits of $N$?

\quad (A) 13 \quad (B) 14 \quad (C) 16 \quad (D) 17 \quad (E) 19

\item Let $O$ be the origin in the coordinate plane and $A$ and $B$ be the two intersection points of the graphs $y=x+4$ and $y=\frac{x^{2}}{2}$. What is the area of $\triangle OAB$?

\quad (A) $6\sqrt{2}$ \quad (B) 10 \quad (C) $\frac{21}{2}$ \quad (D) $\frac{15\sqrt{2}}{2}$ \quad (E) 12

\item Let $x=0.7111111111\dots$ with an infinite number of ones trailing after the seven. The real number $x$ can be written in the form $\frac{m}{n}$ where $m$ and $n$ are relatively prime positive integers. What is $m+n?$

\quad (A) 17 \quad (B) 55 \quad (C) 77 \quad (D) 163 \quad (E) 170

\item Let $ABCD$ be a rhombus with $AC=8$ and $BD=6$. A circle that passes through $A$ is tangent to both segment $\overline{CD}$ and segment $\overline{BC}$. What is the radius of this circle?

\quad (A) $\frac{3}{2}$ \quad (B) 2 \quad (C) $\frac{7}{3}$ \quad (D) $\frac{8}{3}$ \quad (E) 3

\item Let $T$ be the time between 3:00 pm and 4:00 pm on an analog clock such that the minute hand and hour hand coincide. Which of these times is closest to $T$?

\quad (A) 3:15 pm \quad (B) 3:16 pm \quad (C) 3:17 pm \quad (D) 3:18 pm \quad (E) 3:19 pm

\item A stack of 20 cards contains cards labelled with the integers 1 to 10, two of each card. Beth chooses two random cards from the stack without replacement. What is the probability that the nonnegative difference between the numbers on the cards is no more than 3?

\quad (A) $\frac{53}{95}$ \quad (B) $\frac{109}{190}$ \quad (C) $\frac{11}{19}$ \quad (D) $\frac{56}{95}$ \quad (E) $\frac{3}{5}$

\item Triangle $\triangle ABC$ has side lengths $AC=45$ and $BC=53$ and $\angle A=90^{\circ}$. A point is randomly chosen inside the triangle. The probability that the point is closest to $AB$ among the three sides of the triangle is $\frac{m}{n}$ where $m$ and $n$ are relatively prime positive integers. What is $m+n$?

\quad (A) 4 \quad (B) 11 \quad (C) 13 \quad (D) 23 \quad (E) 179

\item Suppose the interior angle measure of a regular $m$-gon is exactly 24 degrees greater than the interior angle measure of a regular $n$-gon. What is the sum of all possible values of $n$?

\quad (A) 38 \quad (B) 39 \quad (C) 42 \quad (D) 45 \quad (E) 47

\item Let $S$ be the set $\{1,2,3,4,5,6,7\}$. How many non-empty subsets of $S$ have a median equal to 4?

\quad (A) 20 \quad (B) 23 \quad (C) 24 \quad (D) 26 \quad (E) 29

\item A rectangle $ABCD$ with side lengths $AB=12$ and $BC=10$ has points $E$ on segment $\overline{AB}$ and $F$ on segment $\overline{CD}$ such that $AE=4$ and $DF=3$. Let $AF$ and $DE$ intersect at $Q$ and let $BF$ and $CE$ intersect at $R$. Suppose line $QR$ intersects $AD$ at point $P$ and intersects $BC$ at $S$. The ratio $PQ:QR:RS$ can be written as $x:y:z$ where $x$, $y$, and $z$ are positive integers and $\gcd(x,y,z)=1$. What is $x+y+z$?

\quad (A) 85 \quad (B) 119 \quad (C) 136 \quad (D) 147 \quad (E) 196

\item What is the sum of all integers $30<x<50$ such that $\text{lcm}(1,2,3,\dots,x-1)\ne \text{lcm}(1,2,3,\dots,x)?$

\quad (A) 199 \quad (B) 215 \quad (C) 264 \quad (D) 280 \quad (E) 301

\item In a $4\times4$ grid of unit squares, 3 unit squares are chosen. Let $R$ be the smallest rectangle whose sides are parallel to the gridlines and fully contain each of the 3 squares in its interior. For how many selections of 3 squares does $R$ have an area of 6?

\quad (A) 84 \quad (B) 96 \quad (C) 120 \quad (D) 132 \quad (E) 156

\item Let $P(x)$ be a third degree polynomial with a leading coefficient of 1 and roots $r$, $s$, and $t$. Suppose that $r+s+t=6$, $rst=1$, and $P(r+s)\cdot P(s+t)\cdot P(r+t)=-104$. What is the value of $P(1)$?

\quad (A) -3 \quad (B) -1 \quad (C) 2 \quad (D) 5 \quad (E) 6

\item Let $d(n)$ denote the number of positive integer divisors of $n$. Call a positive integer $n$ refactorable if $d(n)$ is a divisor of $n$. How many odd refactorable integers are there less than 500?

\quad (A) 3 \quad (B) 4 \quad (C) 5 \quad (D) 6 \quad (E) 7

\item Let $(a_{1},a_{2},\dots,a_{6})$ be a permutation of $(1, 2, \dots, 6)$. There are $N$ permutations for which there is at least one ordered pair of integers $(i, j)$ with $1\le i<j\le6$ that satisfy $a_{i}=j$ and $a_{j}=i$. What is the sum of the digits of $N$?

\quad (A) 9 \quad (B) 11 \quad (C) 12 \quad (D) 13 \quad (E) 15

\item Let $\sigma(n)$ denote the sum of all the positive integer divisors of $n$, including 1 and $n$. How many positive integers $n$ satisfy $\sigma(n)+\sigma(3n)=954$?

\quad (A) 0 \quad (B) 1 \quad (C) 2 \quad (D) 3 \quad (E) 4

\item James and Logan are standing at opposite ends of a long hallway. They both start running at the same time towards each other at different but constant speeds. Logan runs 20% faster than James. When either person reaches a wall at either end of the hallway, he will immediately turn the opposite direction and continue running without changing speed. Define $P_{n}$ to be the point at which James and Logan meet for the $n$th time (possibly at a wall). If the distance between $P_{50}$ and $P_{52}$ is 21, the length of the hallway is $\frac{m}{n}$ , where $m$ and $n$ are relatively prime positive integers. What is $m+n$?

\quad (A) 233 \quad (B) 237 \quad (C) 259 \quad (D) 291 \quad (E) 325

\item Three spheres, $S_{1}$, $S_{2}$, and $S_{3}$ are mutually externally tangent and each are tangent to distinct planes $P_{1}$ and $P_{2}$. Spheres $S_{1}$ and $S_{2}$ both have a radius of 6, and the radius of $S_{3}$ is less than 6. Suppose $P_{1}$ and $P_2$ form a $60^{\circ}$ angle with the three spheres inside the $60^{\circ}$ angle. The radius of $S_{3}$ can be written as $a-b\sqrt{c}$, where $a$, $b$, and $c$ are all positive integers and $c$ is squarefree. What is $a+b+c$?

\quad (A) 21 \quad (B) 22 \quad (C) 23 \quad (D) 24 \quad (E) 25

\end{enumerate}
