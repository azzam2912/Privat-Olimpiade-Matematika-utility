\begin{enumerate}
    \item Let $x$ be a real number such that $\sec x \tan x = 2$. Evaluate $\sec x + \tan x.$
    
    \item Let $0^{\circ}<\theta<45^{\circ}$. Arrange $t_{1}=(\tan \theta)^{\tan \theta}$, $t_{2}=(\tan \theta)^{\cot \theta}$, $t_{3}=(\cot \theta)^{\tan \theta}$, $t_{4}=(\cot \theta)^{\cot \theta},$ in decreasing order.
    
    \item Compute 
    \begin{enumerate}[(a)]
        \item $\sin\dfrac{\pi}{12},\cos\dfrac{\pi}{12};$ and $\tan\dfrac{\pi}{12}$; 
        \item $\cos^{4}\dfrac{\pi}{24}-\sin^{4}\dfrac{\pi}{24}$; 
        \item $\cos 36^{\circ}-\cos 72^{\circ};$ and 
        \item $\sin 10^{\circ}\sin 50^{\circ}\sin 70^{\circ}$
    \end{enumerate}
    
    \item Simplify the expression $\sqrt{\sin^{4}x+4 \cos^{2}x}-\sqrt{\cos^{4}x+4 \sin^{2}x}$.
    
    \item Prove that $1-\cot 23^{\circ}=\dfrac{2}{1-\cot 22^{\circ}}.$
    
    \item Find all $x$ in the interval $(0,\dfrac{\pi}{2})$ such that $\dfrac{\sqrt{3}-1}{\sin x}+\dfrac{\sqrt{3}+1}{\cos x}=4\sqrt{2}.$
    
    \item Region $R$ contains all the points $(x, y)$ such that $x^{2}+y^{2}\le100$ and $\sin(x+y)\ge0$. Find the area of region $R$.
    
    \item In triangle $ABC$, show that $\sin\dfrac{A}{2}\le\dfrac{a}{b+c}$
    
    \item Let $I$ denote the interval $[-\dfrac{\pi}{4},\dfrac{\pi}{4}]$. Determine the function $f$ defined on the interval $[-1, 1]$ such that $f(\sin 2x)=\sin x+\cos x$ and simplify $f(\tan^{2}x)$ for $x$ in the interval $I$.
    
    \item Let $f_{k}(x)=\dfrac{1}{k}(\sin^{k}x+\cos^{k}x)$ for $k=1,2,\dots$ Prove that $f_{4}(x)-f_{6}(x)=\dfrac{1}{12}$ for all real numbers $x$.
    
    \item A circle of radius 1 is randomly placed in a 15-by-36 rectangle $ABCD$ so that the circle lies completely within the rectangle. Given that the probability that the circle will not touch diagonal $AC$ is $\dfrac{m}{n}$ where $m$ and $n$ are relatively prime positive integers, find $m+n$.
    
    \item In triangle $ABC$, $3 \sin A+4 \cos B=6$ and $4 \sin B+3 \cos A=1$. Find the measure of angle $C$.
    \item Prove that $\tan 3a-\tan 2a-\tan a=\tan 3a \tan 2a \tan a$ for all $a\ne\dfrac{k\pi}{2}$ where $k$ is in $\mathbb{Z}$.
    \item Let $a, b, c, d$ be numbers in the interval $[0, \pi]$ such that $\sin a+7 \sin b=4(\sin c+2 \sin d)$, $\cos a+7 \cos b=4(\cos c+2 \cos d)$. Prove that $2 \cos(a-d)=7 \cos(b-c)$.
    \item Express $\sin(x-y)+\sin(y-z)+\sin(z-x)$ as a monomial.
    \item Prove that $(4 \cos^{2}9^{\circ}-3)(4 \cos^{2}27^{\circ}-3)=\tan 9^{\circ}$.
    \item Prove that $(1+\dfrac{a}{\sin x})(1+\dfrac{b}{\cos x})\ge(1+\sqrt{2ab})^{2}$ for all real numbers $a, b, x$ with $a, b\ge0$ and $0<x<\dfrac{\pi}{2}$.
    \item In triangle $ABC$, $\sin A+\sin B+\sin C\le1$. Prove that $\min\{A+B,B+C,C+A\}<30^{\circ}$.
    \item Let $ABC$ be a triangle. Prove that 
    \begin{enumerate}[(a)]
        \item $\tan\dfrac{A}{2}\tan\dfrac{B}{2}+\tan\dfrac{B}{2}\tan\dfrac{C}{2}+\tan\dfrac{C}{2}\tan\dfrac{A}{2}=1;$ 
        \item $\tan\dfrac{A}{2}\tan\dfrac{B}{2}\tan\dfrac{C}{2}\le\dfrac{\sqrt{3}}{9}.$
    \end{enumerate}

    \item Prove that $\dfrac{\tan 3x}{\tan x}=\tan\left(\dfrac{\pi}{3}-x\right)\tan\left(\dfrac{\pi}{3}+x\right)$ for all $x\ne\dfrac{k\pi}{6}$ , where $k$ is in $\mathbb{Z}$.
    \item Given that $(1+\tan 1^{\circ})(1+\tan 2^{\circ})\cdots(1+\tan 45^{\circ})=2^{n}$, find $n$.

    \item In triangle $ABC$, $A-B=120^{\circ}$ and $R=8r$. Find $\cos C$.
    \item Prove that in a triangle $ABC$, $\dfrac{a-b}{a+b}=\tan\dfrac{A-B}{2}\tan\dfrac{C}{2}.$
    \item In triangle $ABC$, $\dfrac{a}{b}=2+\sqrt{3}$ and $\angle C=60^{\circ}$. Find the measure of angles $A$ and $B$.
    \item Let $a, b, c$ be real numbers, all different from $-1$ and $1$, such that $a+b+c= abc$. Prove that $\dfrac{a}{1-a^{2}}+\dfrac{b}{1-b^{2}}+\dfrac{c}{1-c^{2}}=\dfrac{4abc}{(1-a^{2})(1-b^{2})(1-c^{2})}.$
    \item Prove that a triangle $ABC$ is isosceles if and only if $a \cos B+b \cos C+c \cos A=\dfrac{a+b+c}{2}.$
    \item Evaluate $\cos a \cos 2a \cos 3a\cdots \cos 999a,$ where $a=\dfrac{2\pi}{1999}$.
    \item Determine the minimum value of $\dfrac{\sec^{4}\alpha}{\tan^{2}\beta}+\dfrac{\sec^{4}\beta}{\tan^{2}\alpha}$ over all $\alpha, \beta\ne\dfrac{k\pi}{2}$ where $k$ is in $\mathbb{Z}$.
    \item Find all pairs $(x, y)$ of real numbers with $0<x<\dfrac{\pi}{2}$ such that $\dfrac{(\sin x)^{2y}}{(\cos x)^{y^{2}/2}}+\dfrac{(\cos x)^{2y}}{(\sin x)^{y^{2}/2}}=\sin 2x.$
    \item Prove that $\cos 1^{\circ}$ is an irrational number.
    \item Find the maximum value of $S=(1-x_{1})(1-y_{1})+(1-x_{2})(1-y_{2})$ if $x_{1}^{2}+x_{2}^{2}=y_{1}^{2}+y_{2}^{2}=c^{2}$.
    \item Prove that $\dfrac{\sin^{3}a}{\sin b}+\dfrac{\cos^{3}a}{\cos b}\ge \sec(a-b)$ for all $0<a, b<\dfrac{\pi}{2}$.
    \item In triangle $ABC$, show that 
    \begin{enumerate}
        \item $4R=\dfrac{abc}{[ABC]};$ 
        \item $2R^{2}\sin A \sin B \sin C=[ABC]$; 
        \item $2R \sin A \sin B \sin C=r(\sin A+\sin B+\sin C);$ 
        \item $r=4R \sin\dfrac{A}{2}\sin\dfrac{B}{2}\sin\dfrac{C}{2}$; 
        \item $a \cos A+b \cos B+c \cos C=\dfrac{abc}{2R^{2}}.$
    \end{enumerate}
    \item If $\sin \alpha \cos \beta=-\dfrac{1}{2}$ , what are the possible values of $\cos \alpha \sin \beta$?
    \item Let $a, b, c$ be real numbers. Prove that $(ab+bc+ca-1)^{2}\le(a^{2}+1)(b^{2}+1)(c^{2}+1)$.
    \item Prove that $(\sin x+a \cos x)(\sin x+b \cos x)\le1+\left(\dfrac{a+b}{2}\right)^{2}.$
    \item Prove that $|\sin a_{1}|+|\sin a_{2}|+\cdots+|\sin a_{n}|+|\cos(a_{1}+a_{2}+\cdots+a_{n})|\ge1.$
    \item Find all angles $\alpha$ for which the three-element set $S=\{\sin \alpha,\sin 2\alpha,\sin 3\alpha\}$ is equal to the set $T=\{\cos \alpha,\cos 2\alpha,\cos 3\alpha\}.$
   
    \item Let $s$ be the semiperimeter of triangle $ABC$. Prove that 
    \begin{enumerate}
        \item $s=4R \cos\dfrac{A}{2}\cos\dfrac{B}{2}\cos\dfrac{C}{2};$ 
        \item $s\le\dfrac{3\sqrt{3}}{2}R.$
    \end{enumerate}
    \item In triangle $ABC$, show that 
    \begin{enumerate}
        \item $\cos A+\cos B+\cos C=1+4 \sin\dfrac{A}{2}\sin\dfrac{B}{2}\sin\dfrac{C}{2};$ 
        \item $\cos A+\cos B+\cos C\le\dfrac{3}{2}.$
    \end{enumerate}
    \item Let $ABC$ be a triangle. Prove that 
    \begin{enumerate}
        \item $\cos A \cos B \cos C\le\dfrac{1}{8};$ 
        \item $\sin A \sin B \sin C\le\dfrac{3\sqrt{3}}{8};$ 
        \item $\sin A+\sin B+\sin C\le\dfrac{3\sqrt{3}}{2}.$ 
        \item $\cos^{2}A+\cos^{2}B+\cos^{2}C\ge\dfrac{3}{4};$ 
        \item $\sin^{2}A+\sin^{2}B+\sin^{2}C\le\dfrac{9}{4};$ 
        \item $\cos 2A+\cos 2B+\cos 2C\ge-\dfrac{3}{2};$ 
        \item $\sin 2A+\sin 2B+\sin 2C\le\dfrac{3\sqrt{3}}{2}.$
    \end{enumerate}
    \item Let $\{T_{n}(x)\}_{n=0}^{\infty}$ be the sequence of polynomials such that $T_{0}(x)=1$, $T_{1}(x)= x$, $T_{i+1}=2xT_{i}(x)-T_{i-1}(x)$ for all positive integers $i$. The polynomial $T_{n}(x)$ is called the $n$th Chebyshev polynomial. 
    \begin{enumerate}
        \item Prove that $T_{2n+1}(x)$ and $T_{2n}(x)$ are odd and even functions, respectively; 
        \item Prove that $T_{n}(\cos \theta)=\cos(n\theta)$ for all nonnegative integers $n$; 
        \item Prove that for real numbers $x$ in the interval $[-1, 1]$, $-1\le T_{n}(x)\le1$; 
        \item Prove that $T_{n+1}(x)>T_{n}(x)$ for real numbers $x$ with $x>1$; 
        \item Determine all the roots of $T_{n}(x)$; 
        \item Determine all the roots of $P_{n}(x)=T_{n}(x)-1$.
    \end{enumerate}
    \item Let $ABC$ be a triangle with $\angle BAC=40^{\circ}$ and $\angle ABC=60^{\circ}$. Let $D$ and $E$ be the points lying on the sides $AC$ and $AB$, respectively, such that $\angle CBD=40^{\circ}$ and $\angle BCE=70^{\circ}.$ Segments $BD$ and $CE$ meet at $F$. Show that $AF \perp BC$.
    \item Let $S$ be an interior point of triangle $ABC$. Show that at least one of $\angle SAB$, $\angle SBC$, and $\angle SCA$ is less than or equal to $30^{\circ}$.
    \item Let $a=\dfrac{\pi}{7}$ . 
    \begin{enumerate}
        \item Show that $\sin^{2}3a-\sin^{2}a=\sin 2a \sin 3a$; 
        \item Show that $\csc a=\csc 2a+\csc 4a$; 
        \item Evaluate $\cos a-\cos 2a+\cos 3a$; 
        \item Prove that $\cos a$ is a root of the equation $8x^{3}+4x^{2}-4x-1=0$; 
        \item Prove that $\cos a$ is irrational; 
        \item Evaluate $\tan a \tan 2a \tan 3a$; 
        \item Evaluate $\tan^{2}a+\tan^{2}2a+\tan^{2}3a;$ 
        \item Evaluate $\tan^{2}a \tan^{2}2a+\tan^{2}2a \tan^{2}3a+\tan^{2}3a \tan^{2}a$; 
        \item Evaluate $\cot^{2}a+\cot^{2}2a+\cot^{2}3a$.
    \end{enumerate}

    \item Two exercises on $\sin k^{\circ}\sin(k+1)^{\circ};$ 
    \begin{enumerate}
        \item Find the smallest positive integer $n$ such that $\dfrac{1}{\sin 45^{\circ}\sin 46^{\circ}}+\dfrac{1}{\sin 47^{\circ}\sin 48^{\circ}}+\cdots+\dfrac{1}{\sin 133^{\circ}\sin 134^{\circ}}=\dfrac{1}{\sin n^{\circ}}$;.
        \item Prove that $\dfrac{1}{\sin 1^{\circ}\sin 2^{\circ}}+\dfrac{1}{\sin 2^{\circ}\sin 3^{\circ}}+\cdots+\dfrac{1}{\sin 89^{\circ}\sin 90^{\circ}}=\dfrac{\cos 1^{\circ}}{\sin^{2}1^{\circ}}$.
    \end{enumerate}
    
    \item Let $x, y, z$ be real numbers with $x\ge y\ge z\ge\dfrac{\pi}{12}$ such that $x+y+z=\dfrac{\pi}{2}$ . Find the maximum and the minimum values of the product $\cos x \sin y \cos z$.

    \item Find the sum of all $x$ in the interval $[0,2\pi]$ such that $3 \cot^{2}x+8 \cot x+3=0$.

    \item Compute the sums $\binom{n}{1}\sin a+\binom{n}{2}\sin 2a+\cdots+\binom{n}{n}\sin na$ and $\binom{n}{1}\cos a+\binom{n}{2}\cos 2a+\cdots+\binom{n}{n}\cos na.$
    
    \item Find the minimum value of $|\sin x+\cos x+\tan x+\cot x+\sec x+\csc x|$ for real numbers $x$.

    
    \item Let $a, b, c$ be real numbers such that $\sin a+\sin b+\sin c\ge\dfrac{3}{2}.$ Prove that $\sin\left(a-\dfrac{\pi}{6}\right)+\sin\left(b-\dfrac{\pi}{6}\right)+\sin\left(c-\dfrac{\pi}{6}\right)\ge0$
    
    \item Consider any four numbers in the interval $\left[\dfrac{\sqrt{2}-\sqrt{6}}{2},\dfrac{\sqrt{2}+\sqrt{6}}{2}\right]$. Prove that there are two of them, say $a$ and $b$, such that $\left|a\sqrt{4-b^{2}}-b\sqrt{4-a^{2}}\right|\le2.$
    
    \item Let $a$ and $b$ be real numbers in the interval $\left[0,\dfrac{\pi}{2}\right]$. Prove that $\sin^{6}a+3 \sin^{2}a \cos^{2}b+\cos^{6}b=1$ if and only if $a=b$.
    
    
    \item For a triangle $XYZ$, let $r_{XYZ}$ denote its inradius. Given that the convex pentagon $ABCDE$ is inscribed in a circle, prove that if $r_{ABC}=r_{AED}$ and $r_{ABD}=r_{AEC}$, then triangles $ABC$ and $AED$ are congruent.


    \item Prove that the average of the numbers $2 \sin 2^{\circ}$, $4 \sin 4^{\circ}$, $6 \sin 6^{\circ}, \dots, 180 \sin 180^{\circ}$ is $s \cot 1^{\circ}$.

    \item Triangle $ABC$ has the following property: there is an interior point $P$ such that $\angle PAB=10^{\circ}$, $\angle PBA=20^{\circ}$, $\angle PCA=30^{\circ}$, and $\angle PAC=40^{\circ}$. Prove that triangle $ABC$ is isosceles.
\end{enumerate}
